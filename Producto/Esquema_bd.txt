CREATE TABLE public.sede (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  nombre text NOT NULL UNIQUE,
  CONSTRAINT sede_pkey PRIMARY KEY (id)
);
CREATE TABLE public.usuario (
  id uuid NOT NULL,
  nombre text,
  created_at timestamp without time zone DEFAULT now(),
  email text NOT NULL DEFAULT 'NOT NULL'::text UNIQUE,
  rol USER-DEFINED DEFAULT 'jugador'::rol_usuario,
  telefono text UNIQUE,
  rut text,
  foto_url text,
  CONSTRAINT usuario_pkey PRIMARY KEY (id),
  CONSTRAINT usuario_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.clase (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  sede_id uuid,
  cupo_maximo integer DEFAULT 15,
  created_at timestamp without time zone DEFAULT now(),
  titulo text DEFAULT 'Clase'::text,
  descripcion text,
  profesor_id uuid,
  fecha_hora timestamp without time zone,
  CONSTRAINT clase_pkey PRIMARY KEY (id),
  CONSTRAINT clase_sede_id_fkey FOREIGN KEY (sede_id) REFERENCES public.sede(id),
  CONSTRAINT clase_profesor_id_fkey FOREIGN KEY (profesor_id) REFERENCES public.usuario(id)
);
CREATE TABLE public.clase_usuario (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  usuario_id uuid,
  created_at timestamp without time zone DEFAULT now(),
  asistencia USER-DEFINED DEFAULT 'sin_confirmar'::asistencia,
  clase_id uuid,
  CONSTRAINT clase_usuario_pkey PRIMARY KEY (id),
  CONSTRAINT clase_usuario_usuario_id_fkey FOREIGN KEY (usuario_id) REFERENCES public.usuario(id),
  CONSTRAINT clase_usuario_clase_id_fkey FOREIGN KEY (clase_id) REFERENCES public.clase(id)
);
CREATE TABLE public.plan (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  nombre text NOT NULL UNIQUE,
  tokens_mensuales integer NOT NULL,
  precio integer,
  created_at timestamp without time zone DEFAULT now(),
  CONSTRAINT plan_pkey PRIMARY KEY (id)
);
CREATE TABLE public.boleta (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  usuario_id uuid NOT NULL,
  estado text NOT NULL CHECK (estado = ANY (ARRAY['pendiente'::text, 'pagado'::text, 'rechazado'::text, 'anulado'::text])),
  total integer NOT NULL DEFAULT 0,
  transaccion_id text,
  created_at timestamp without time zone DEFAULT now(),
  updated_at timestamp without time zone DEFAULT now(),
  recurrencia_id uuid,
  flow_confirmada boolean DEFAULT false,
  CONSTRAINT boleta_pkey PRIMARY KEY (id),
  CONSTRAINT boleta_usuario_id_fkey FOREIGN KEY (usuario_id) REFERENCES public.usuario(id),
  CONSTRAINT boleta_recurrencia_id_fkey FOREIGN KEY (recurrencia_id) REFERENCES public.recurrencia(id)
);

  CONSTRAINT membresia_usuario_id_fkey FOREIGN KEY (usuario_id) REFERENCES public.usuario(id),

CREATE TABLE public.producto (
